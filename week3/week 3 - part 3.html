    for name, lat, lng in zip(names, latitudes, longitudes):
        print(name)

            CLIENT_ID, 
            CLIENT_SECRET, 
            VERSION, 
            lat,

            radius, 
            LIMIT)
            
        # make the GET request
        results = requests.get(url).json()["response"]['groups'][0]['items']

        venues_list.append([(
            name,

def return_most_common_venues(row, num_top_venues):
    row_categories = row.iloc[1:]
